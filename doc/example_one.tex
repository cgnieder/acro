\AcroNewTemplate[list]{physics}{%
  \def\TableCode{}%
  \acronymsmapT{%
    \AcroPutRight\TableCode{%
      \def\AcronymID{##1}%
      \acrowrite{short}%
      &
      \def\AcronymID{##1}%
      \acrowrite{unit}%
      &
      \def\AcronymID{##1}%
      \acrowrite{long}%
      &
      \def\AcronymID{##1}%
      \acroifpagesT{\acropages{S. }{S. }}%
      \tabularnewline
    }%
  }%
  \acroheading
  \acropreamble
  \par\noindent
  \setlength\LTleft{0pt}%
  \setlength\LTright{0pt}%
  \begin{longtable}{@{}lll@{\extracolsep{\fill}}l@{}}
    \TableCode
  \end{longtable}
}
