\NewDocumentCommand \DeclareAcroProperty {st?t!t|t>m}
  {
    \acro_property_declare:en
      {
        \IfBooleanT {#1} {unique=true,}
        \IfBooleanT {#2} {boolean=true,}
        \IfBooleanT {#3} {mandatory=true,}
        \IfBooleanT {#4} {static=true,}
        \IfBooleanT {#5} {display=true}
      }
      {#6}
  }
